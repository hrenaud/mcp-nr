```python
"""
Audit URL — analyse technique d'un site web avec EcoIndex et mapping GreenIT.
"""
from collections import Counter
from urllib.parse import urlparse

# ============================================================================
# ECOINDEX
# Source: https://github.com/cnumr/ecoindex_js + https://github.com/cnumr/ecoindex_reference
# ============================================================================

QUANTILES_DOM = [
    0, 47, 75, 159, 233, 298, 358, 417, 476, 537, 603, 674,
    753, 843, 949, 1076, 1237, 1459, 1801, 2479, 594601,
]
QUANTILES_REQ = [
    0, 2, 15, 25, 34, 42, 49, 56, 63, 70, 78, 86, 95,
    105, 117, 130, 147, 170, 205, 281, 3920,
]
QUANTILES_SIZE = [
    0, 1.37, 144.7, 319.53, 479.46, 631.97, 783.38, 937.91,
    1098.62, 1265.47, 1448.32, 1648.27, 1876.08, 2142.06,
    2465.37, 2866.31, 3401.59, 4155.73, 5400.08, 8037.54, 223212.26,
]

# Sorted descending by value threshold
GRADES = [
    {"value": 80, "grade": "A"},
    {"value": 70, "grade": "B"},
    {"value": 55, "grade": "C"},
    {"value": 40, "grade": "D"},
    {"value": 25, "grade": "E"},
    {"value": 10, "grade": "F"},
    {"value": 0,  "grade": "G"},
]


def compute_quantile(quantiles: list, value: float) -> float:
    """Interpolated quantile position for a value in a quantile array."""
    for i in range(1, len(quantiles)):
        if value < quantiles[i]:
            return (i - 1) + (value - quantiles[i - 1]) / (quantiles[i] - quantiles[i - 1])
    return float(len(quantiles) - 1)


def compute_ecoindex(dom: int, req: int, size_kb: float) -> dict:
    """
    Compute EcoIndex score (0-100) and grade (A-G).

    Args:
        dom:     Number of DOM nodes
        req:     Number of HTTP requests
        size_kb: Total transferred size in kilobytes

    Returns:
        {"score": float, "grade": str}
    """
    q_dom  = compute_quantile(QUANTILES_DOM, dom)
    q_req  = compute_quantile(QUANTILES_REQ, req)
    q_size = compute_quantile(QUANTILES_SIZE, size_kb)
    score  = 100 - 5 * (3 * q_dom + 2 * q_req + q_size) / 6
    score  = max(0.0, min(100.0, score))
    grade  = "G"
    for g in GRADES:
        if score > g["value"]:
            grade = g["grade"]
            break
    return {"score": round(score, 2), "grade": grade}
```

```python
# ============================================================================
# GREENIT MAPPER
# ============================================================================

_TRACKING_PATTERNS = [
    ("google_analytics", ["google-analytics.com", "analytics.js", "gtag"]),
    ("gtm", ["googletagmanager.com", "gtm.js"]),
    ("facebook_pixel", ["connect.facebook.net", "fbevents.js"]),
    ("hotjar", ["hotjar.com"]),
    ("intercom", ["intercom.io"]),
]


def detect_tracking(urls: list) -> list:
    """Detect known tracking/analytics scripts in a list of request URLs."""
    detected = []
    for name, patterns in _TRACKING_PATTERNS:
        if any(p in u for u in urls for p in patterns):
            detected.append(name)
    return detected


def map_metrics_to_fiches(metrics: dict, cache: dict) -> list:
    """
    Return relevant GreenIT best-practice sheets for the given page metrics.

    Args:
        metrics: Page metrics dict (dom_nodes, requests, size_kb, ecoindex, …)
        cache:   GreenIT fiches cache from charger_cache()

    Returns:
        Up to 20 fiches sorted by combined score descending.
    """
    target_resources = set()

    if metrics.get("images_sans_lazy", 0) > 0:
        target_resources.update(["network", "images"])
    if metrics.get("dom_nodes", 0) > 1500:
        target_resources.add("cpu")
    if metrics.get("requests", 0) > 40:
        target_resources.update(["requests", "network"])
    if metrics.get("size_kb", 0) > 1024:
        target_resources.update(["network", "storage"])
    if metrics.get("tracking_scripts"):
        target_resources.update(["javascript", "cpu"])
    if not metrics.get("compression", True):
        target_resources.update(["network", "server"])
    if not metrics.get("cache_headers", True):
        target_resources.update(["storage", "server"])
    if len(metrics.get("fonts", [])) > 2:
        target_resources.update(["network", "css"])
    if metrics.get("iframes", 0) > 0:
        target_resources.update(["javascript", "network"])

    include_prioritaires = metrics.get("ecoindex", {}).get("grade", "A") in ("D", "E", "F", "G")

    seen = set()
    resultats = []

    for fiche_id, fiche in cache.items():
        if fiche_id in seen:
            continue
        saved = set(fiche.get("saved_resources", []))
        ei = fiche.get("environmental_impact", 0) or 0
        pi = fiche.get("priority_implementation", 0) or 0

        match = bool(saved & target_resources)
        prioritaire = include_prioritaires and ei >= 4 and pi >= 4

        if match or prioritaire:
            resultats.append({
                "id": fiche_id,
                "titre": fiche.get("title"),
                "environmental_impact": ei,
                "priority_implementation": pi,
                "url": fiche.get("url", ""),
            })
            seen.add(fiche_id)

    resultats.sort(key=lambda x: x["environmental_impact"] + x["priority_implementation"], reverse=True)
    return resultats[:20]
```

```python
# ============================================================================
# REPORT BUILDER
# ============================================================================


def build_report(pages: list, start_url: str, cache: dict, format: str = "markdown"):
    """
    Build an audit report from crawled page metrics.

    Args:
        pages:     List of page metrics dicts from crawl()
        start_url: The URL that was audited
        cache:     GreenIT fiches cache from charger_cache()
        format:    "markdown" or "json"

    Returns:
        str (markdown) or dict (json)
    """
    valid_pages = [p for p in pages if "error" not in p]

    # Global EcoIndex: average score → recompute grade from averages
    scores = [p["ecoindex_score"] for p in valid_pages if "ecoindex_score" in p]
    if scores:
        ecoindex_moyen = round(sum(scores) / len(scores), 2)
        avg_dom  = int(sum(p["dom_nodes"] for p in valid_pages) / len(valid_pages))
        avg_req  = int(sum(p["requests"] for p in valid_pages) / len(valid_pages))
        avg_size = sum(p["size_kb"] for p in valid_pages) / len(valid_pages)
        grade_moyen = compute_ecoindex(avg_dom, avg_req, avg_size)["grade"]
    else:
        ecoindex_moyen = None
        grade_moyen = None

    # Global recommendations: count fiche occurrences across pages
    fiche_counter = Counter()
    fiche_map = {}
    for page in valid_pages:
        page_ecoindex = {"grade": page.get("grade", "A")}
        metrics_for_mapper = dict(page, ecoindex=page_ecoindex)
        for f in map_metrics_to_fiches(metrics_for_mapper, cache):
            fiche_counter[f["id"]] += 1
            fiche_map[f["id"]] = f

    recommandations = [
        {**fiche_map[fid], "occurrences": count}
        for fid, count in fiche_counter.most_common(10)
    ]

    # Pages enriched with fiches
    pages_out = []
    for page in valid_pages:
        p = {k: v for k, v in page.items() if k != "_links"}
        page_ecoindex = {"grade": page.get("grade", "A")}
        p["fiches_greenit"] = map_metrics_to_fiches(dict(page, ecoindex=page_ecoindex), cache)
        pages_out.append(p)

    # Also include error pages in output
    for page in pages:
        if "error" in page:
            pages_out.append(page)

    data = {
        "url": start_url,
        "pages_analysees": len(pages),
        "ecoindex_moyen": ecoindex_moyen,
        "grade_moyen": grade_moyen,
        "pages": pages_out,
        "recommandations_globales": recommandations,
    }

    if format == "json":
        return data

    # Markdown
    lines = [
        f"# Audit GreenIT — {start_url}",
        "",
        f"**Pages analysées :** {len(pages)}  ",
        f"**EcoIndex moyen :** {ecoindex_moyen} / 100  ",
        f"**Grade moyen :** {grade_moyen}",
        "",
        "---",
        "",
    ]

    for page in pages_out:
        if "error" in page:
            lines.append(f"## {page['url']} ⚠️ Erreur")
            lines.append(f"> {page['error']}")
            lines.append("")
            continue

        lines.append(f"## {page['url']}")
        lines.append("")
        lines.append(f"**EcoIndex :** {page.get('ecoindex_score')} / 100 — Grade **{page.get('grade')}**")
        lines.append("")
        lines.append("| Métrique | Valeur |")
        lines.append("|---|---|")
        lines.append(f"| Nœuds DOM | {page.get('dom_nodes')} |")
        lines.append(f"| Requêtes | {page.get('requests')} |")
        lines.append(f"| Poids total | {page.get('size_kb')} KB |")
        lines.append(f"| Images sans lazy | {page.get('images_sans_lazy')} |")
        lines.append(f"| Compression | {'✓' if page.get('compression') else '✗'} |")
        lines.append(f"| Cache headers | {'✓' if page.get('cache_headers') else '✗'} |")
        lines.append(f"| Scripts tracking | {', '.join(page.get('tracking_scripts', [])) or 'aucun'} |")
        lines.append(f"| TTFB | {page.get('ttfb_ms')} ms |")
        lines.append(f"| FCP | {page.get('fcp_ms')} ms |")
        lines.append("")

        fiches = page.get("fiches_greenit", [])
        if fiches:
            lines.append("**Fiches GreenIT applicables :**")
            for f in fiches[:5]:
                lines.append(
                    f"- [{f['titre']}]({f['url']}) "
                    f"(impact: {f['environmental_impact']}, priorité: {f['priority_implementation']})"
                )
        lines.append("")

    lines += [
        "---",
        "",
        "## Recommandations globales",
        "",
    ]
    for r in recommandations:
        lines.append(
            f"- **{r['titre']}** — {r['occurrences']} page(s) · "
            f"impact: {r['environmental_impact']}, priorité: {r['priority_implementation']}"
        )

    return "\n".join(lines)
```

```python
async def extract_page_metrics(page, url: str) -> dict:
    """
    Navigate to url with a Playwright page and extract all technical metrics.

    Args:
        page: Playwright Page object (already created by caller)
        url:  URL to navigate to

    Returns:
        Metrics dict. Includes "_links" key (internal links found on page).
    """
    requests_data = []
    total_bytes = 0

    def on_response(response):
        nonlocal total_bytes
        try:
            cl = response.headers.get("content-length")
            if cl:
                total_bytes += int(cl)
        except Exception:
            pass
        requests_data.append({
            "url": response.url,
            "resource_type": response.request.resource_type,
            "headers": dict(response.headers),
        })

    page.on("response", on_response)

    try:
        await page.goto(url, timeout=30000, wait_until="networkidle")
    except Exception:
        await page.goto(url, timeout=30000, wait_until="load")

    dom_metrics = await page.evaluate("""() => {
        const imgs = Array.from(document.querySelectorAll('img'));
        const fonts = [...new Set(
            Array.from(document.styleSheets)
                .flatMap(s => { try { return Array.from(s.cssRules); } catch(e) { return []; } })
                .filter(r => r instanceof CSSFontFaceRule)
                .map(r => r.style.getPropertyValue('font-family').replace(/['"]/g,'').trim())
                .filter(Boolean)
        )];
        return {
            dom_nodes: document.querySelectorAll('*').length,
            images_sans_lazy: imgs.filter(i => i.getAttribute('loading') !== 'lazy').length,
            images_sans_alt: imgs.filter(i => !i.hasAttribute('alt')).length,
            iframes: document.querySelectorAll('iframe').length,
            fonts: fonts,
            links: Array.from(document.querySelectorAll('a[href]'))
                       .map(a => a.href)
                       .filter(h => h.startsWith('http')),
        };
    }""")

    perf = await page.evaluate("""() => {
        const nav = performance.getEntriesByType('navigation')[0] || {};
        const paint = {};
        performance.getEntriesByType('paint').forEach(e => { paint[e.name] = Math.round(e.startTime); });
        return {
            ttfb_ms: nav.responseStart ? Math.round(nav.responseStart - nav.requestStart) : null,
            fcp_ms: paint['first-contentful-paint'] || null,
        };
    }""")

    html_resp = next((r for r in requests_data if r["resource_type"] == "document"), None)
    compression = False
    cache_headers = False
    if html_resp:
        h = html_resp["headers"]
        enc = h.get("content-encoding", "")
        compression = "gzip" in enc or "br" in enc
        cache_headers = bool(h.get("cache-control") or h.get("etag"))

    all_urls = [r["url"] for r in requests_data]
    size_kb = round(total_bytes / 1024, 2) if total_bytes > 0 else 0
    nb_req = len(requests_data)
    ecoindex = compute_ecoindex(dom_metrics["dom_nodes"], nb_req, size_kb)

    return {
        "url": url,
        "dom_nodes": dom_metrics["dom_nodes"],
        "requests": nb_req,
        "size_kb": size_kb,
        "resource_types": dict(Counter(r["resource_type"] for r in requests_data)),
        "images_sans_lazy": dom_metrics["images_sans_lazy"],
        "images_sans_alt": dom_metrics["images_sans_alt"],
        "iframes": dom_metrics["iframes"],
        "fonts": dom_metrics["fonts"],
        "compression": compression,
        "cache_headers": cache_headers,
        "tracking_scripts": detect_tracking(all_urls),
        "ttfb_ms": perf["ttfb_ms"],
        "fcp_ms": perf["fcp_ms"],
        "lcp_ms": None,
        "ecoindex_score": ecoindex["score"],
        "grade": ecoindex["grade"],
        "_links": dom_metrics["links"],
    }
```

```python
# ============================================================================
# CRAWLER
# ============================================================================


async def crawl(start_url: str, max_pages: int = 5) -> list:
    """
    Crawl a website starting from start_url, staying on the same domain.

    Args:
        start_url: URL to start crawling from
        max_pages: Maximum number of pages to visit (default: 5)

    Returns:
        List of page metrics dicts (one per visited page).
    """
    from playwright.async_api import async_playwright

    base_domain = urlparse(start_url).netloc
    visited = set()
    queue = [start_url]
    results = []

    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        context = await browser.new_context(
            user_agent="Mozilla/5.0 (compatible; GreenIT-Auditor/1.0)"
        )

        while queue and len(visited) < max_pages:
            url = queue.pop(0)
            if url in visited:
                continue
            visited.add(url)

            page = await context.new_page()
            try:
                metrics = await extract_page_metrics(page, url)
                links = metrics.pop("_links", [])
                results.append(metrics)

                # Enqueue same-domain links not yet visited
                for link in links:
                    parsed = urlparse(link)
                    clean = parsed._replace(fragment="").geturl()
                    if parsed.netloc == base_domain and clean not in visited and clean not in queue:
                        queue.append(clean)
            except Exception as e:
                results.append({"url": url, "error": str(e)})
            finally:
                await page.close()

        await context.close()
        await browser.close()

    return results
```
